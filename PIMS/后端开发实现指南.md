### 1. JWT工具类核心实现

```java
@Component
public class JwtUtil {
    @Value("${jwt.secret}")
    private String secret;
    
    @Value("${jwt.expiration}")
    private Long expiration;
    
    // 生成Token
    public String generateToken(Long userId, String userType) {
        Map<String, Object> claims = new HashMap<>();
        claims.put("userId", userId);
        claims.put("userType", userType);
        
        return Jwts.builder()
            .setClaims(claims)
            .setSubject(userId.toString())
            .setIssuedAt(new Date())
            .setExpiration(new Date(System.currentTimeMillis() + expiration))
            .signWith(Keys.hmacShaKeyFor(secret.getBytes()), SignatureAlgorithm.HS256)
            .compact();
    }
    
    // 解析Token
    public Claims parseToken(String token) {
        return Jwts.parserBuilder()
            .setSigningKey(Keys.hmacShaKeyFor(secret.getBytes()))
            .build()
            .parseClaimsJws(token)
            .getBody();
    }
}
```

### 2. 验证码工具类核心实现

```java
@Component
@Slf4j
public class VerifyCodeUtil {
    
    // 生成6位数字验证码
    public String generateCode() {
        return String.format("%06d", new Random().nextInt(1000000));
    }
    
    // 发送验证码（模拟，打印到日志）
    public void sendCode(String phone, String code) {
        log.info("===== 验证码发送 =====");
        log.info("手机号：{}", phone);
        log.info("验证码：{}", code);
        log.info("======================");
    }
    
    // 验证验证码（简化实现，实际应该使用Redis存储）
    private Map<String, CodeInfo> codeCache = new ConcurrentHashMap<>();
    
    public boolean verify(String phone, String code) {
        CodeInfo codeInfo = codeCache.get(phone);
        if (codeInfo == null) {
            return false;
        }
        if (System.currentTimeMillis() > codeInfo.getExpireTime()) {
            codeCache.remove(phone);
            return false;
        }
        return codeInfo.getCode().equals(code);
    }
}
```

### 3. 用户注册Service核心实现

```java
@Override
public void register(RegisterDTO dto) {
    // 1. 验证手机号格式
    if (!PhoneValidator.isValid(dto.getPhone())) {
        throw new BusinessException(ResultCode.PARAM_ERROR.getCode(), "手机号格式不正确");
    }
    
    // 2. 验证房号格式
    if (!RoomValidator.isValid(dto.getRoomNumber())) {
        throw new BusinessException(ResultCode.PARAM_ERROR.getCode(), "房号格式不正确（格式：Xxxx，如A310）");
    }
    
    // 3. 检查手机号是否已注册
    if (userMapper.selectOne(new LambdaQueryWrapper<User>()
            .eq(User::getPhone, dto.getPhone())) != null) {
        throw new BusinessException(ResultCode.PHONE_EXISTS);
    }
    
    // 4. 检查房号是否已存在
    if (userMapper.selectOne(new LambdaQueryWrapper<User>()
            .eq(User::getRoomNumber, dto.getRoomNumber())) != null) {
        throw new BusinessException(ResultCode.ROOM_NUMBER_EXISTS);
    }
    
    // 5. 验证验证码
    if (!verifyCodeUtil.verify(dto.getPhone(), dto.getVerifyCode())) {
        throw new BusinessException(ResultCode.VERIFY_CODE_ERROR);
    }
    
    // 6. 创建用户
    User user = new User();
    user.setPhone(dto.getPhone());
    user.setPassword(passwordEncoder.encode(dto.getPassword()));
    user.setRoomNumber(dto.getRoomNumber());
    user.setStatus(0);  // 未入住
    
    userMapper.insert(user);
}
```

### 4. 入住审核Service核心实现

```java
@Override
@Transactional(rollbackFor = Exception.class)
public void processAudit(Long auditId, Integer status, BigDecimal pricePerSqm, Long adminId) {
    // 1. 查询审核记录
    AuditRecord auditRecord = auditRecordMapper.selectById(auditId);
    if (auditRecord == null) {
        throw new BusinessException(ResultCode.NOT_FOUND.getCode(), "审核记录不存在");
    }
    
    // 2. 更新审核记录
    auditRecord.setStatus(status);
    auditRecord.setAdminId(adminId);
    auditRecord.setAuditTime(new Date());
    auditRecordMapper.updateById(auditRecord);
    
    // 3. 如果审核通过，创建住户信息
    if (status == 1) {
        Resident resident = new Resident();
        resident.setUserId(auditRecord.getUserId());
        resident.setName(auditRecord.getName());
        resident.setIdCard(auditRecord.getIdCard());
        resident.setPhone(auditRecord.getPhone());
        resident.setRoomNumber(auditRecord.getRoomNumber());
        resident.setArea(auditRecord.getArea());
        resident.setPricePerSqm(pricePerSqm != null ? pricePerSqm : new BigDecimal(Constants.DEFAULT_PRICE_PER_SQM));
        resident.setStatus(1);  // 已入住
        resident.setRegisterTime(new Date());
        residentMapper.insert(resident);
        
        // 更新用户状态
        User user = new User();
        user.setId(auditRecord.getUserId());
        user.setStatus(1);
        userMapper.updateById(user);
    }
}
```

### 5. 缴费提醒定时任务核心实现

```java
@Component
@Slf4j
public class PaymentReminderTask {
    
    @Autowired
    private ResidentMapper residentMapper;
    
    @Autowired
    private PaymentRecordMapper paymentRecordMapper;
    
    @Autowired
    private MessageMapper messageMapper;
    
    /**
     * 每天凌晨1点执行
     */
    @Scheduled(cron = "0 0 1 * * ?")
    public void sendPaymentReminder() {
        log.info("开始执行缴费提醒任务...");
        
        // 查询所有已入住的住户
        List<Resident> residents = residentMapper.selectList(
            new LambdaQueryWrapper<Resident>().eq(Resident::getStatus, 1)
        );
        
        Date now = new Date();
        Calendar calendar = Calendar.getInstance();
        
        for (Resident resident : residents) {
            // 计算缴费日（注册日期的异月同日）
            calendar.setTime(resident.getRegisterTime());
            int paymentDay = calendar.get(Calendar.DAY_OF_MONTH);
            
            calendar.setTime(now);
            int currentDay = calendar.get(Calendar.DAY_OF_MONTH);
            
            // 提前5天发送提醒
            if (currentDay == paymentDay - 5) {
                // 检查本月是否已缴费
                String currentMonth = new SimpleDateFormat("yyyy-MM").format(now);
                PaymentRecord paymentRecord = paymentRecordMapper.selectOne(
                    new LambdaQueryWrapper<PaymentRecord>()
                        .eq(PaymentRecord::getUserId, resident.getUserId())
                        .eq(PaymentRecord::getPaymentMonth, currentMonth)
                );
                
                // 未缴费则发送提醒
                if (paymentRecord == null) {
                    BigDecimal amount = resident.getArea().multiply(resident.getPricePerSqm());
                    String content = String.format("尊敬的住户，您本月即将缴纳物业费，金额为%.2f元，请及时缴费，如已完成缴费请忽略", amount);
                    
                    Message message = new Message();
                    message.setUserId(resident.getUserId());
                    message.setTitle("物业费缴费提醒");
                    message.setContent(content);
                    message.setIsRead(0);
                    message.setMessageType(1);
                    messageMapper.insert(message);
                    
                    log.info("已向用户{}发送缴费提醒", resident.getUserId());
                }
            }
        }
        
        log.info("缴费提醒任务执行完成");
    }
}
```
